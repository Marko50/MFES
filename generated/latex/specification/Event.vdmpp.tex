 public DaysOfMonth : nat * nat -> nat
 DaysOfMonth(y, m) == 
  if (m = 2) then (
   if isLeapYear(y) then 29
   else 28
  )
  else if (m = 4 or m = 6 or m = 9 or m = 11) then 30
  else 31;
 
(*@
\label{isLeapYear:94}
@*)
 public isLeapYear : nat -> bool
 isLeapYear (y) ==
  y mod 4 = 0 and y mod 100 <> 0 or y mod 400 = 0;
